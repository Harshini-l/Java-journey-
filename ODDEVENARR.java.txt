package oddevenarr;
import java.util.Scanner;
public class ODDEVENARRAY {
	public static void main(String[]args) {
	int odd=0;
	int even=0;
    int n =7;
	int arr[]= {5, 7, 12, 16, 2, 6, 9};
	for(int i=0;i<n;i++) {
		if(arr[i]%2==0) {
			even++;
			System.out.println("Even:");
		}
		else {
			odd++;
			System.out.println("Odd:");
		}
	}

}
}
