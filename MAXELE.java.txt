package maxelementarr;
import java.util.Scanner;
public class MAXELE {
	public static void main(String[]args) {
		Scanner sc=new Scanner(System.in);
		int n=sc.nextInt();
		
		int a[]= {17,15,18,11};
		int max=0;
		for(int i=0; i<n; i++) {
			if(a[i]>max) {
				max=a[i];
			}
		}
     System.out.println(max);
}
}
